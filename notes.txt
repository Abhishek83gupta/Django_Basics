// Add below code
from django.db import models
 
class Item(models.Model):
    name = models.CharField(max_length=100)
    description = models.TextField()
 
    def __str__(self):
        return self.name


// Generate the database based on model provided

python manage.py makemigrations    // create a file in migrations folder
